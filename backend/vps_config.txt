# Rate limiting zone
limit_req_zone $binary_remote_addr zone=api:10m rate=30r/s;

server {
    server_name api.youposhmaroc.com;

    # Servir les uploads directement (images produits)
    location /uploads/ {
        alias /var/www/youposh/backend/uploads/;
        expires 30d;
        add_header Cache-Control "public, max-age=2592000, immutable";
        access_log off;
    }

    # Cache pour les fichiers statiques
    location ~* \.(jpg|jpeg|png|gif|webp|svg|ico|css|js)$ {
        expires 30d;
        add_header Cache-Control "public, max-age=2592000, immutable";
        add_header Vary Accept-Encoding;
        access_log off;
    }

    # API avec rate limiting
    location / {
        limit_req zone=api burst=50 nodelay;

        proxy_pass http://127.0.0.1:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        client_max_body_size 50M;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/api.youposhmaroc.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/api.youposhmaroc.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}

server {
    if ($host = api.youposhmaroc.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    listen 80;
    server_name api.youposhmaroc.com;
    return 404; # managed by Certbot
}
